Time: O(N) Each element is added and removed at most once. Linear pass.
Space: O(1) Only variables for sum and max are stored. No extra arrays.

func maxSum(arr []int, k int) int { maxSum, windowSum := 0, 0 for i := 0; i < k; i++ { windowSum += arr[i] } maxSum = windowSum for i := k; i < len(arr); i++ { windowSum += arr[i] - arr[i-k] if windowSum > maxSum { maxSum = windowSum } } return maxSum }
public int maxSum(int[] arr, int k) { int maxSum = 0, windowSum = 0; for (int i = 0; i < k; i++) { windowSum += arr[i]; } maxSum = windowSum; for (int i = k; i < arr.length; i++) { windowSum += arr[i] - arr[i-k]; maxSum = Math.max(maxSum, windowSum); } return maxSum; }
