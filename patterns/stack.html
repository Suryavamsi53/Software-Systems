ðŸ§  2. Core Concept Theory
Last In, First Out

A Stack is a collection that supports pushing elements onto the top and popping them from the top. It's like a stack of plates.

Use Cases:

  • Undo/Redo: Storing history.
  • Parsing: Validating syntax (JSON, XML, code).
  • DFS: Depth-First Search uses a stack (explicit or recursion).
ðŸ§© 5. Canonical Amazon Problem
Valid Parentheses (LeetCode 20)

Problem: Given a string s containing just the characters '(', ')', '{', '}', '[' and ']', determine if the input string is valid.

Input: s = "()[]{}"
Output: true

Why Amazon Loves It: It's a classic parsing problem that tests if you understand how to pair elements using LIFO logic.

func isValid(s string) bool {
    // Map for closing -> opening
    pairs := map[rune]rune{
        ')': '(',
        ']': '[',
        '}': '{',
    }
    
    stack := []rune{}
    
    for _, char := range s {
        // If it's a closing bracket
        if open, isClose := pairs[char]; isClose {
            // Check if stack is empty or top doesn't match
            if len(stack) == 0 || stack[len(stack)-1] != open {
                return false
            }
            // Pop
            stack = stack[:len(stack)-1]
        } else {
            // It's an opening bracket, push
            stack = append(stack, char)
        }
    }
    
    // Valid only if stack is empty
    return len(stack) == 0
}

class Solution {
    public boolean isValid(String s) {
        Stack stack = new Stack<>();
        
        for (char c : s.toCharArray()) {
            if (c == '(') stack.push(')');
            else if (c == '{') stack.push('}');
            else if (c == '[') stack.push(']');
            else if (stack.isEmpty() || stack.pop() != c) return false;
        }
        
        return stack.isEmpty();
    }
}

class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        mapping = {")": "(", "}": "{", "]": "["}
        
        for char in s:
            if char in mapping:
                # Closing bracket
                top_element = stack.pop() if stack else '#'
                if mapping[char] != top_element:
                    return False
            else:
                # Opening bracket
                stack.append(char)
        
        return not stack
