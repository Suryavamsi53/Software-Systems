func twoPointers(nums []int) int {
    left, right := 0, len(nums)-1
    for left < right {
        if condition(left, right) {
            return result
        } else if tooSmall {
            left++
        } else {
            right--
        }
    }
    return -1
}
Sliding Window (Dynamic) ‚ñº 
func slidingWindow(s string) int {
    left := 0
    for right := 0; right < len(s); right++ {
        // 1. Add s[right] to window
        
        // 2. Shrink window while condition is invalid/met
        for condition() {
            // Remove s[left]
            left++
        }
        // 3. Update result
    }
    return result
}
Binary Search ‚ñº 
func binarySearch(nums []int, target int) int {
    left, right := 0, len(nums)-1
    for left <= right {
        mid := left + (right-left)/2
        if nums[mid] == target {
            return mid
        } else if nums[mid] < target {
            left = mid + 1
        } else {
            right = mid - 1
        }
    }
    return -1
}

DFS (Recursive) ‚ñº 
func dfs(node *Node, visited map[*Node]bool) {
    if node == nil || visited[node] {
        return
    }
    visited[node] = true
    // Process node
    for _, neighbor := range node.Neighbors {
        dfs(neighbor, visited)
    }
}
BFS (Queue) ‚ñº 
func bfs(start *Node) {
    queue := []*Node{start}
    visited := map[*Node]bool{start: true}
    
    for len(queue) > 0 {
        node := queue[0]
        queue = queue[1:]
        // Process node
        
        for _, neighbor := range node.Neighbors {
            if !visited[neighbor] {
                visited[neighbor] = true
                queue = append(queue, neighbor)
            }
        }
    }
}

Backtracking ‚ñº 
func backtrack(path []int, options []int) {
    if isSolution(path) {
        result = append(result, path)
        return
    }
    for i := 0; i < len(options); i++ {
        // Choose
        path = append(path, options[i])
        // Explore
        backtrack(path, options)
        // Unchoose
        path = path[:len(path)-1]
    }
}
